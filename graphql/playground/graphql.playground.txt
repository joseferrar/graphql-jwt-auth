Graphql Operations:-

mutation Mutation($registerInput: RegisterInput) {
   registerUser(registerInput:$registerInput) {
     email
     password
     username
   }
 }

 mutation Mutation($loginInput: LoginInput) {
   loginUser(loginInput:$loginInput) {
     email
     password
   }
 }

query {
  allUsers {
    username
    email
  }
}
